自动获取Nexus4售卖状态的Python脚本

使用pyquery做爬虫解析GooglePlay页面 获取状态 然后通过Gmail发给指定邮箱 每隔一个小时自动检测一次 使用 nohup python *.py & 可在后台运行

```python

coding=utf-8

easy_install pyquery OR sudo apt-get install python-pyquery

from pyquery import PyQuery as pq import smtplib from email.Message import Message import time

发给谁

mailto_list = "xxx@xx.xx"

设置服务器 用户名 口令 邮箱后缀

mail_host = "smtp.gmail.com" mail_user = "xxx@gmail.com" mail_pass = "you password"

def sendMail(sub, content): ''' sub:主题 content:内容 ''' msg = Message() msg['Subject'] = sub msg['From'] = mail_user msg['To'] = mailto_list msg.set_payload(content) try: s = smtplib.SMTP(mail_host, port=587, timeout=200) s.starttls() s.login(mail_user, mail_pass) s.sendmail(mail_user, mailto_list, msg.as_string()) return True except Exception, e: print str(e) return False

def checkSoldState(): doc = pq(url = r'https://play.google.com/store/devices/details?id=nexus_4_16gb') soldMsg = doc('span').filter('.hardware-price-description').text() print soldMsg if soldMsg != 'Sold out': print 'sent mail' sendMail('Nexus4 Info', 'Sold out')

def say(): print "hello"

main

while True: checkSoldState() time.sleep(3600) ```pyhon
